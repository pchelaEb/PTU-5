{Исходные данные:}
koy=0.15;T1=0.5;T2=0.8;Ty=0.05;k1=8;kp=0.8;kn=2;
kg=1;Tg=0.01;

W1(s)=k1/(Ty*s+1);
W2(s)=kp;
W3(s)=kn/s;
Woy(s)=koy/((T1*s+1)*(T2*s+1));
Wg(s)=kg/(Tg*s+1);

{Передаточная функция регулятора:}
Wr(s)=W1(s)*(W2(s)+W3(s));

{Передаточная функция разомкнутой системы:}
W(s)=Wr(s)*Woy(s)*Wg(s);

{Передаточная функция замкнутой системы:}
F(s)=Wr(s)*Woy(s)/(1+W(s));
